#!/usr/bin/env python3
from __future__ import annotations

import re
import subprocess
import sys
from pathlib import Path

REPO = Path("/Users/hof/Dropbox/Privat/GitHub/agentic-project-kit")
EXPECTED_BRANCH = "feature/b11-transfer-report-contract-semantics"
TRANSFER_FILE = ".agentic/transfer/inbox/b11_transfer_report_contract_semantics_apply.py.txt"
EXPECTED_DIRTY = {
    "src/agentic_project_kit/cli_commands/transfer.py",
    "src/agentic_project_kit/transfer_uplink.py",
    "tests/test_transfer_uplink.py",
}


def run(cmd: list[str], *, check: bool = True) -> subprocess.CompletedProcess[str]:
    print("$ " + " ".join(cmd), flush=True)
    completed = subprocess.run(cmd, cwd=REPO, text=True, capture_output=True, check=False)
    if completed.stdout:
        print(completed.stdout, end="")
    if completed.stderr:
        print(completed.stderr, end="", file=sys.stderr)
    if check and completed.returncode != 0:
        raise SystemExit(f"COMMAND_FAILED returncode={completed.returncode}: {' '.join(cmd)}")
    return completed


def read(rel: str) -> str:
    return (REPO / rel).read_text(encoding="utf-8")


def write(rel: str, text: str) -> None:
    (REPO / rel).write_text(text, encoding="utf-8")


def branch() -> str:
    return run(["git", "branch", "--show-current"]).stdout.strip()


def dirty_files() -> list[str]:
    out = run(["git", "status", "--short"]).stdout.splitlines()
    return sorted(line[3:].strip() for line in out if line.strip())


def require_clean_feature_branch() -> None:
    print("### PRECHECK: branch and clean worktree ###")
    current = branch()
    print(f"CURRENT_BRANCH={current}")
    if current != EXPECTED_BRANCH:
        raise SystemExit(f"WRONG_BRANCH expected={EXPECTED_BRANCH} actual={current}")
    dirty = dirty_files()
    print("DIRTY_FILES=" + " ".join(dirty))
    if dirty:
        raise SystemExit("WORKTREE_NOT_CLEAN: stop and inspect before patching")


def require_no_existing_product_diff() -> None:
    run(["git", "fetch", "origin", "main"])
    diff_lines = run(["git", "diff", "--name-only", "origin/main...HEAD"], check=False).stdout.splitlines()
    product_diff = [line for line in diff_lines if line != TRANSFER_FILE]
    print("EXISTING_BRANCH_DIFF=" + (" ".join(diff_lines) if diff_lines else "none"))
    print("EXISTING_PRODUCT_DIFF=" + (" ".join(product_diff) if product_diff else "none"))
    if product_diff:
        raise SystemExit("BRANCH_ALREADY_HAS_PRODUCT_DIFF: inspect before applying this patch")


def patch_transfer_uplink() -> None:
    rel = "src/agentic_project_kit/transfer_uplink.py"
    s = read(rel)

    if "local_report_path: str" not in s:
        s = s.replace(
            "    remote_report_path: str\n    transfer_upload: str\n",
            "    remote_report_path: str\n    transfer_upload: str\n    local_report_path: str = \"\"\n    transfer_report_written: str = \"done\"\n",
            1,
        )

    if '"local_report_path": self.local_report_path' not in s:
        s = s.replace(
            '            "remote_report_path": self.remote_report_path,\n            "transfer_upload": self.transfer_upload,\n',
            '            "remote_report_path": self.remote_report_path,\n            "transfer_upload": self.transfer_upload,\n            "local_report_path": self.local_report_path,\n            "transfer_report_written": self.transfer_report_written,\n',
            1,
        )

    old_summary = (
        '            "### SUMMARY ###",\n'
        '            "TRANSFER_UPLOAD=done",\n'
        '            f"REMOTE_REPORT={result.remote_report_path}",\n'
        '            f"FINAL_SIGNAL={result.final_signal}",\n'
        '            f"FINAL_NEXT={result.next_action}",\n'
        '            f"CHAT_REPLY={result.chat_reply}",\n'
    )
    new_summary = (
        '            "### SUMMARY ###",\n'
        '            "TRANSFER_REPORT_WRITTEN=done",\n'
        '            f"LOCAL_REPORT={result.local_report_path or result.remote_report_path}",\n'
        '            f"FINAL_SIGNAL={result.final_signal}",\n'
        '            "FINAL_NEXT=Run transfer publish-last-report",\n'
        '            "CHAT_REPLY=d | NEXT=Run transfer publish-last-report",\n'
    )
    if old_summary in s:
        s = s.replace(old_summary, new_summary, 1)

    s = re.sub(
        r'(remote_report_path=str\(timestamped_json\),\n\s*transfer_upload="done",)(?!\n\s*local_report_path=)',
        r'\1\n        local_report_path=str(timestamped_json),\n        transfer_report_written="done",',
        s,
    )

    s = s.replace(
        'chat_reply="g",\n        next_action=next_action,',
        'chat_reply="d | NEXT=Run transfer publish-last-report",\n        next_action="Run transfer publish-last-report",',
        2,
    )

    write(rel, s)


def patch_transfer_cli() -> None:
    rel = "src/agentic_project_kit/cli_commands/transfer.py"
    s = read(rel)
    s = s.replace(
        '        typer.echo("TRANSFER_UPLOAD=done")\n'
        '        typer.echo(f"REMOTE_REPORT={result.remote_report_path}")\n'
        '        typer.echo("CHAT_REPLY=g")',
        '        typer.echo("TRANSFER_REPORT_WRITTEN=done")\n'
        '        typer.echo(f"LOCAL_REPORT={result.local_report_path or result.remote_report_path}")\n'
        '        typer.echo("CHAT_REPLY=d | NEXT=Run transfer publish-last-report")',
        2,
    )
    write(rel, s)


def patch_tests() -> None:
    rel = "tests/test_transfer_uplink.py"
    s = read(rel)

    s = s.replace('assert result.chat_reply == "g"', 'assert result.chat_reply == "d | NEXT=Run transfer publish-last-report"', 4)
    s = s.replace(
        'assert "CHAT_REPLY=g" in (tmp_path / LATEST_LOG).read_text(encoding="utf-8")',
        'assert "CHAT_REPLY=d | NEXT=Run transfer publish-last-report" in (tmp_path / LATEST_LOG).read_text(encoding="utf-8")',
    )
    s = s.replace(
        'assert "TRANSFER_UPLOAD=done" in log\n    assert "REMOTE_REPORT=docs/reports/transfer_runs/" in log\n    assert "CHAT_REPLY=g" in log',
        'assert "TRANSFER_REPORT_WRITTEN=done" in log\n    assert "LOCAL_REPORT=docs/reports/transfer_runs/" in log\n    assert "CHAT_REPLY=d | NEXT=Run transfer publish-last-report" in log',
    )
    s = s.replace(
        'assert lines[0] == "TRANSFER_UPLOAD=done"\n    assert lines[1].startswith("REMOTE_REPORT=docs/reports/transfer_runs/")\n    assert lines[2] == "CHAT_REPLY=g"',
        'assert lines[0] == "TRANSFER_REPORT_WRITTEN=done"\n    assert lines[1].startswith("LOCAL_REPORT=docs/reports/transfer_runs/")\n    assert lines[2] == "CHAT_REPLY=d | NEXT=Run transfer publish-last-report"',
        2,
    )
    s = s.replace('assert data["transfer_upload"] == "done"', 'assert data["transfer_report_written"] == "done"', 1)
    s = s.replace('assert data["chat_reply"] == "g"', 'assert data["chat_reply"] == "d | NEXT=Run transfer publish-last-report"', 1)

    s = s.replace(
        'assert lines[0] == "TRANSFER_REPORT_WRITTEN=done"\n    assert lines[1].startswith("LOCAL_REPORT=docs/reports/terminal/transfer_handoff_reports/")\n    assert lines[2] == "CHAT_REPLY=d | NEXT=Run transfer publish-last-report"',
        'assert lines[0] == "TRANSFER_UPLOAD=done"\n    assert lines[1].startswith("REMOTE_REPORT=docs/reports/terminal/transfer_handoff_reports/")\n    assert lines[2] == "CHAT_REPLY=g"',
    )
    s = s.replace(
        'assert published["chat_reply"] == "d | NEXT=Run transfer publish-last-report"',
        'assert published["chat_reply"] == "g"',
    )
    write(rel, s)


def run_checks() -> None:
    run([str(REPO / ".venv/bin/python"), "-m", "py_compile",
         "src/agentic_project_kit/transfer_uplink.py",
         "src/agentic_project_kit/cli_commands/transfer.py"])
    run([str(REPO / ".venv/bin/python"), "-m", "pytest", "-q",
         "tests/test_transfer_uplink.py",
         "tests/test_transfer_repo_actions.py",
         "tests/test_transfer_pr_actions.py"])
    run([str(REPO / ".venv/bin/ruff"), "check",
         "src/agentic_project_kit/transfer_uplink.py",
         "src/agentic_project_kit/cli_commands/transfer.py",
         "tests/test_transfer_uplink.py"])


def verify_dirty() -> None:
    dirty = set(dirty_files())
    print("DIRTY_FILES=" + " ".join(sorted(dirty)))
    if dirty != EXPECTED_DIRTY:
        raise SystemExit(f"UNEXPECTED_DIRTY_FILES expected={sorted(EXPECTED_DIRTY)} actual={sorted(dirty)}")
    run(["git", "diff", "--stat"])
    run(["git", "diff", "--name-only"])


def main() -> None:
    require_clean_feature_branch()
    require_no_existing_product_diff()
    print("### APPLY CONTRACT SEMANTICS PATCH ###")
    patch_transfer_uplink()
    patch_transfer_cli()
    patch_tests()
    verify_dirty()
    run_checks()
    print("### FINAL STATUS ###")
    run(["git", "branch", "--show-current"])
    run(["git", "status", "-sb"])
    print("RESULT=PATCH_READY")
    print("NEXT=Review diff, then commit explicitly on feature branch. No push/PR was performed.")


if __name__ == "__main__":
    main()
