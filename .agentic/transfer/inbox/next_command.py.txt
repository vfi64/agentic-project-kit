def _workflow_state(completed: subprocess.CompletedProcess[str], blocking_reason: str) -> str:
    if completed.returncode == 0:
        return "SAFE"
    if completed.returncode == 2 or blocking_reason:
        return "BLOCKED"
    return "FAILED"

def _resolve_stash_entries() -> tuple[list[tuple[str, str, str]], subprocess.CompletedProcess[str]]:
    command = ["git", "stash", "list", "--format=%gd%x09%H%x09%gs"]
    completed = _run(command)
    entries: list[tuple[str, str, str]] = []
    if completed.returncode == 0:
        for line in completed.stdout.splitlines():
            parts = line.split("\t", 2)
            if len(parts) == 3:
                entries.append((parts[0], parts[1], parts[2]))
    return entries, completed


def _stash_show_paths(stash_object_id: str) -> list[str]:
    completed = _run(["git", "stash", "show", "--name-only", stash_object_id])
    if completed.returncode != 0:
        return []
    return [line.strip() for line in completed.stdout.splitlines() if line.strip()]


def stash_inspect(stash_object_id: str) -> RepoActionResult:
    entries, listed = _resolve_stash_entries()
    if listed.returncode != 0:
        return _result("stash-inspect", listed.args, listed, "Inspect stash list failure before continuing.")
    matches = [(ref, oid, subject) for ref, oid, subject in entries if oid == stash_object_id]
    if len(matches) != 1:
        completed = subprocess.CompletedProcess(
            ["git", "stash", "show", "--name-status", stash_object_id],
            2,
            listed.stdout,
            f"Expected exactly one stash object id match for {stash_object_id}, found {len(matches)}.\n",
        )
        return _result("stash-inspect", completed.args, completed, "Re-read stash list and use one full stash object id.", stash_object_id=stash_object_id, blocking_reason="stash_object_id_not_unique")
    ref, oid, subject = matches[0]
    show = _run(["git", "stash", "show", "--name-status", oid])
    paths = _stash_show_paths(oid)
    stdout = f"STASH_REF={ref}\nSTASH_OBJECT_ID={oid}\nSTASH_SUBJECT={subject}\n" + show.stdout
    completed = subprocess.CompletedProcess(show.args, show.returncode, stdout, show.stderr)
    return _result("stash-inspect", show.args, completed, "Use stash-drop-safe only with this full object id if the stash is obsolete.", stash_ref=ref, stash_object_id=oid, stash_subject=subject, affected_paths=paths)


def stash_drop_safe(stash_object_id: str, *, expected_subject: str = "", expected_paths: list[str] | None = None) -> RepoActionResult:
    if len(stash_object_id) != 40:
        completed = subprocess.CompletedProcess(
            ["git", "stash", "drop", stash_object_id],
            2,
            "",
            "Refusing destructive stash drop without a full 40-character stash object id.\n",
        )
        return _result("stash-drop-safe", completed.args, completed, "Run transfer stash-list and stash-inspect first.", stash_object_id=stash_object_id, blocking_reason="stash_object_id_not_full")
    entries, listed = _resolve_stash_entries()
    if listed.returncode != 0:
        return _result("stash-drop-safe", listed.args, listed, "Inspect stash list failure before continuing.")
    matches = [(ref, oid, subject) for ref, oid, subject in entries if oid == stash_object_id]
    if len(matches) != 1:
        completed = subprocess.CompletedProcess(
            ["git", "stash", "drop", stash_object_id],
            2,
            listed.stdout,
            f"Refusing stash drop: expected exactly one stash object id match for {stash_object_id}, found {len(matches)}.\n",
        )
        return _result("stash-drop-safe", completed.args, completed, "Re-read stash list; stash refs may have changed.", stash_object_id=stash_object_id, blocking_reason="stash_object_id_not_unique")
    ref, oid, subject = matches[0]
    if expected_subject and expected_subject not in subject:
        completed = subprocess.CompletedProcess(
            ["git", "stash", "drop", ref],
            2,
            listed.stdout,
            f"Refusing stash drop: expected subject fragment {expected_subject!r} not found in {subject!r}.\n",
        )
        return _result("stash-drop-safe", completed.args, completed, "Inspect stash subject mismatch before destructive action.", stash_ref=ref, stash_object_id=oid, stash_subject=subject, blocking_reason="expected_subject_mismatch")
    paths = _stash_show_paths(oid)
    expected = list(expected_paths or [])
    if expected and sorted(paths) != sorted(expected):
        completed = subprocess.CompletedProcess(
            ["git", "stash", "drop", ref],
            2,
            "\n".join(paths) + "\n",
            "Refusing stash drop: affected paths do not match expected paths.\n",
        )
        return _result("stash-drop-safe", completed.args, completed, "Inspect stash path mismatch before destructive action.", stash_ref=ref, stash_object_id=oid, stash_subject=subject, affected_paths=paths, blocking_reason="expected_paths_mismatch")
    command = ["git", "stash", "drop", ref]
    dropped = _run(command)
    return _result("stash-drop-safe", command, dropped, "Run transfer stash-list and repo-status to verify cleanup.", stash_ref=ref, stash_object_id=oid, stash_subject=subject, affected_paths=paths)
