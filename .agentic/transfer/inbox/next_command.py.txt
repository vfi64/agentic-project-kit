from __future__ import annotations

import json
import subprocess
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

ROOT = Path.cwd()
OUTBOX = ROOT / ".agentic/transfer/outbox/last_result.txt"
TARGET_BRANCH = "feature/compact-transfer-report-output"


def run(command: list[str]) -> dict[str, Any]:
    completed = subprocess.run(command, cwd=ROOT, text=True, capture_output=True, check=False)
    return {
        "command": command,
        "returncode": completed.returncode,
        "stdout": completed.stdout,
        "stderr": completed.stderr,
        "ok": completed.returncode == 0,
    }


def git_value(*args: str) -> str:
    result = run(["git", *args])
    return str(result["stdout"]).strip() if result["ok"] else ""


def main() -> int:
    steps: list[dict[str, Any]] = []
    blockers: list[str] = []
    initial_status = git_value("status", "--short")
    if initial_status:
        blockers.append("dirty_worktree_before_transfer_start")

    for name, command in [
        ("sync-main", ["./.venv/bin/agentic-kit", "transfer", "sync-main"]),
        ("rules-acknowledge", ["./.venv/bin/agentic-kit", "rules", "acknowledge"]),
        ("remote-work-start", ["./.venv/bin/agentic-kit", "transfer", "remote-work-start", TARGET_BRANCH]),
    ]:
        if blockers:
            break
        step = run(command)
        step["name"] = name
        steps.append(step)
        if not step["ok"]:
            blockers.append(name + "_failed")

    final_branch = git_value("branch", "--show-current")
    final_status = git_value("status", "--short")
    ok = not blockers and final_branch == TARGET_BRANCH and not final_status
    if not blockers and final_branch != TARGET_BRANCH:
        blockers.append("wrong_final_branch")
    if not blockers and final_status:
        blockers.append("dirty_worktree_after_transfer_start")

    payload = {
        "schema_version": 1,
        "kind": "local_to_llm_last_result",
        "action": "start_compact_transfer_report_output_slice",
        "target_branch": TARGET_BRANCH,
        "finished_at": datetime.now(timezone.utc).isoformat(),
        "final_branch": final_branch,
        "steps": steps,
        "blockers": blockers,
        "result_status": "PASS" if ok else "BLOCKED",
        "final_signal": "d" if ok else "f",
        "next_action": "Ready for implementation transfer file." if ok else "Inspect blockers before continuing.",
        "chat_reply": "g | NEXT=Ready for implementation transfer file." if ok else "f | NEXT=Inspect blockers before continuing.",
    }
    OUTBOX.parent.mkdir(parents=True, exist_ok=True)
    OUTBOX.write_text(json.dumps(payload, indent=2, sort_keys=True) + "\n", encoding="utf-8")

    print("TRANSFER_PAYLOAD=start_compact_transfer_report_output_slice")
    print(f"STATE={payload['result_status']}")
    print(f"BRANCH={final_branch}")
    print(f"BLOCKERS={len(blockers)}")
    print(f"FINAL_SIGNAL={payload['final_signal']}")
    print("FINAL_NEXT=" + str(payload["next_action"]))
    print("CHAT_REPLY=" + str(payload["chat_reply"]))
    return 0 if ok else 1


if __name__ == "__main__":
    raise SystemExit(main())
