#!/usr/bin/env python3
from pathlib import Path
import subprocess, sys

ROOT = Path.cwd()
EXPECTED_BRANCH = "feature/b11-transfer-report-contract-semantics"
EXPECTED = {
    "src/agentic_project_kit/cli_commands/transfer.py",
    "src/agentic_project_kit/transfer_uplink.py",
    "tests/test_transfer_uplink.py",
}

def sh(cmd, check=True):
    print("$ " + " ".join(cmd), flush=True)
    p = subprocess.run(cmd, cwd=ROOT, text=True, capture_output=True)
    if p.stdout: print(p.stdout, end="")
    if p.stderr: print(p.stderr, end="", file=sys.stderr)
    if check and p.returncode:
        raise SystemExit(f"COMMAND_FAILED {p.returncode}: {' '.join(cmd)}")
    return p.stdout

def read(p): return (ROOT / p).read_text(encoding="utf-8")
def write(p, s): (ROOT / p).write_text(s, encoding="utf-8")
def dirty():
    return sorted(line[3:] for line in sh(["git", "status", "--short"]).splitlines() if line.strip())

print("### PRECHECK ###")
branch = sh(["git", "branch", "--show-current"]).strip()
print("CURRENT_BRANCH=" + branch)
if branch != EXPECTED_BRANCH:
    raise SystemExit("WRONG_BRANCH")
unexpected = set(dirty()) - EXPECTED
if unexpected:
    raise SystemExit("UNEXPECTED_DIRTY=" + " ".join(sorted(unexpected)))

print("### REPAIR transfer_uplink next_action semantics ###")
p = "src/agentic_project_kit/transfer_uplink.py"
s = read(p)
s = s.replace(
    'chat_reply="d | NEXT=Run transfer publish-last-report",\n        next_action="Run transfer publish-last-report",',
    'chat_reply="d | NEXT=Run transfer publish-last-report",\n        next_action=next_action,',
)
write(p, s)

print("### REPAIR show-last-report expectation ###")
p = "tests/test_transfer_uplink.py"
s = read(p)
s = s.replace('assert data["chat_reply"] == "g"', 'assert data["chat_reply"] == "d | NEXT=Run transfer publish-last-report"', 1)
write(p, s)

print("### VERIFY EXPECTED DIRTY FILES ###")
d = set(dirty())
print("DIRTY_FILES=" + " ".join(sorted(d)))
if d != EXPECTED:
    raise SystemExit("UNEXPECTED_DIRTY_SET")

print("### CHECKS ###")
sh([".venv/bin/python", "-m", "py_compile", "src/agentic_project_kit/transfer_uplink.py", "src/agentic_project_kit/cli_commands/transfer.py"])
sh([".venv/bin/python", "-m", "pytest", "-q", "tests/test_transfer_uplink.py", "tests/test_transfer_repo_actions.py", "tests/test_transfer_pr_actions.py"])
sh([".venv/bin/ruff", "check", "src/agentic_project_kit/transfer_uplink.py", "src/agentic_project_kit/cli_commands/transfer.py", "tests/test_transfer_uplink.py"])
sh(["git", "diff", "--stat"])
print("RESULT=PATCH_READY")
print("NEXT=Review diff and commit the three expected files only.")
